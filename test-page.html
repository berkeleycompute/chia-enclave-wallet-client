Option 1: Create a React App for Testing

1 Create a new React app or use an existing one:

2 Install your Chia wallet client package:

# If testing locally

3 Replace src/App.js with this code: 
import React from 'react';
import { ChiaWalletTestPage } from 'chia-enclave-wallet-client';

function App() {
  // Replace with your actual JWT token
  const jwtToken = "your_jwt_token_here";
  
  return (
    <div className="App">
      <ChiaWalletTestPage jwtToken={jwtToken} />
    </div>
  );
}

export default App;
4 Start the React app:

Option 2: Direct ChiaWalletButton Usage

1 Use the ChiaWalletButton component directly: 
import React from 'react';
import { ChiaWalletButton } from 'chia-enclave-wallet-client';

function MyApp() {
  const handleWalletUpdate = (walletData) => {
    console.log('Wallet updated:', walletData);
  };

  return (
    <div>
      <h1>My Chia Wallet App</h1>
      <ChiaWalletButton
        jwtToken="your_jwt_token_here"
        variant="primary"
        size="large"
        onWalletUpdate={handleWalletUpdate}
      />
    </div>
  );
}

export default MyApp;

Option 3: Individual Dialog Hooks

1 Use individual dialog hooks for custom UI: 
import React from 'react';
import { 
  DialogProvider, 
  useSendFundsDialog, 
  useMakeOfferDialog,
  useReceiveFundsDialog 
} from 'chia-enclave-wallet-client';

function WalletActions() {
  const sendFunds = useSendFundsDialog();
  const makeOffer = useMakeOfferDialog();
  const receiveFunds = useReceiveFundsDialog();

  return (
    <div>
      <button onClick={sendFunds.open}>
        üí∏ Send Funds
      </button>
      
      <button onClick={receiveFunds.open}>
        üí∞ Receive Funds
      </button>
      
      <button onClick={() => makeOffer.open(nftData)}>
        ü§ù Make NFT Offer
      </button>
      
      <p>Send dialog open: {sendFunds.isOpen ? 'Yes' : 'No'}</p>
    </div>
  );
}

function App() {
  return (
    <DialogProvider>
      <WalletActions />
    </DialogProvider>
  );
}

export default App;
üéØ Key Points:
  • The DialogProvider must wrap your components that use dialog hooks
  • All dialog state is automatically managed by the hooks
  • Modals render automatically when hooks are used
  • Each dialog has an isOpen state, open() and close() methods
  • NFT-related dialogs have additional selectedNft state
